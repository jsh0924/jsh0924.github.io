## 1\. 변수

실수 > 정수  
실수형인 float에 값을 대입할 때는 숫자의 맨 뒤에 f를 붙여야 함.  
붙이지 않을 경우 8바이트인 double로 인식한다.  
Int a = 100, b = 200; 으로 요약 가능.  
  
````java
	int a;
	float b;

	a = (int) 123.45f; // 강제 형변. 소수는 사라진다.
	b = 200;
````

<!-- !국민취업지원제도 참여 기간! 수급자격 인증받은 날부터! -->

````java
package com.study;
// 원의 넓이와 둘레를 구하는 프로그램 구현
// 원의 넓이 : 반지름 * 반지름 * 3.141592
// 원의 둘레 : 반지름 * 2 * 3.141592
// 반지름 : 10, a = 넓이, b = 둘레
// 반지름이 ~인 원의 넓이는 ~이고
// 반지름이 ~인 원의 둘레는 ~이다
public class Test007 {

	public static void main(String[] args) {
		double half = 10;
		double a = 0;
		double b = 0;
		double pi = 3.141592;
		
		a = half * half * pi;
		b = half * 2 * pi;
		
		System.out.println("반지름이 " + half + "인 원의 넓이는 " + a + "이고, \n반지름이 " + half + "인 원의 둘레는 " + b + "이다.");
		System.out.printf("반지름이 %.0f인 원의 넓이는 %f이고, \n반지름이 %.0f인 원의 둘레는 %f이다.", half, a, half, b);
	}
}

````
